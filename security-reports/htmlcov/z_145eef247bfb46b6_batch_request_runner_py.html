26 print(f"BatchRequestRunner: Starting batch run with {len(self.requests)} requests")

28 

29 for i, request in enumerate(self.requests): 

30 print(f"BatchRequestRunner: Running request {i+1}/{len(self.requests)}: {request.name}") 

31 

32 # Convert request to dict format 

33 request_data = { 

34 "method": request.method, 

35 "url": request.url, 

36 "headers": request.headers, 

37 "params": request.params, 

38 "body": request.body, 

39 "body_type": request.body_type, 

40 "pre_request_script": request.pre_request_script, 

41 "tests": request.tests 

42 } 

43 print(f"DEBUG: Request data passed to RequestRunner: {request_data}") # Debug print

47 runner = RequestRunner(request_data, self.environment)

51 runner.start()

61 test_results = {"passed": False, "results": ["No response received"]} 

62 

63 print(f"BatchRequestRunner: Request {request.name} completed with status {response.get('status_code', 'Unknown')}") 

64 

65 except Exception as e:

83 # Small delay between requests (reduced for faster execution)
